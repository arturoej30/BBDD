-- Ejemplos de trabajo con fechas, y volcado de valores a variables
DECLARE
	hora varchar2(2);
	minuto varchar2(2);
	segundo varchar2(10);
	
BEGIN

	SELECT extract(hour from SYSTIMESTAMP), 
	extract(minute from SYSTIMESTAMP),
	extract(second from SYSTIMESTAMP) 
	INTO hora, minuto, segundo 
	FROM dual;
		
	dbms_output.put_line('Son las ' || hora || ':' || minuto || ':' || segundo);

END;
/
